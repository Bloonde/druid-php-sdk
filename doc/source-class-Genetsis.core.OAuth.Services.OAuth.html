142     /**
143      * Gets an "access_token" for the current web client.
144      *
145      * @param string $endpoint_url The endpoint where "access_token" is requested.
146      * @param string $code The authorization code returned by Genetsis ID.
147      * @param string $redirect_url Where the user will be redirected.
148      * @return array An array with the following data:
149      *      [
150      *          'access_token' => An instance of {@link AccessToken}
151      *          'refresh_token' => An instance of {@link RefreshToken}
152      *          'login_status' => An instance of {@link LoginStatus}
153      *      ]
154      * @throws \Exception If there is an error.

157     public function doGetAccessToken ($endpoint_url, $code, $redirect_url)
158     {
159         try {
160             if (($endpoint_url = trim(( string )$endpoint_url)) == '') {
161                 throw new Exception ('Endpoint URL is empty');
162             }
163             if (($code = trim(( string )$code)) == '') {
164                 throw new Exception ('Code is empty');
165             }
166             if (($redirect_url = trim(( string )$redirect_url)) == '') {
167                 throw new Exception ('Redirect URL is empty');
168             }
169 
170             $params = [];
171             $params['grant_type'] = AuthMethodsCollection::GRANT_TYPE_AUTH_CODE;
172             $params['code'] = $code;
173             $params['redirect_uri'] = $redirect_url;
174             $params['client_id'] = $this->getConfig()->getClientId();
175             $params['client_secret'] = $this->getConfig()->getClientSecret();
176             $response = SC::getHttpService()->execute($endpoint_url, $params, HttpMethodsCollection::POST);
177 
178             $this->checkErrors($response);
179 
180             if (!isset ($response ['result']->access_token) || ($response ['result']->access_token == '')) {
181                 throw new Exception ('The access_token retrieved is empty');
182             }
183             if (!isset ($response ['result']->refresh_token) || ($response ['result']->refresh_token == '')) {
184                 throw new Exception ('The refresh_token retrieved is empty');
185             }
186 
187             $expires_in = self::DEFAULT_EXPIRES_IN;
188             if (isset ($response['result']->expires_in)) {
189                 $expires_in = intval($response['result']->expires_in);
190             }
191             $expires_in = ($expires_in - ($expires_in * self::SAFETY_RANGE_EXPIRES_IN));
192             $expires_at = (time() + $expires_in);
193             $refresh_expires_at = ($expires_at + (60*60*24*12));
194 
195             $result['access_token'] = new AccessToken(trim($response ['result']->access_token), $expires_in, $expires_at, '/');
196             $result['refresh_token'] = new RefreshToken(trim($response ['result']->refresh_token), 0, $refresh_expires_at, '/');
197 
198             $this->storeToken($result['access_token']);
199             $this->storeToken($result['refresh_token']);
200 
201             $loginStatus = new LoginStatus();
202             if (isset($response['result']->login_status)) {
203                 $loginStatus->setCkusid(isset($response['result']->login_status->uid) ? $response['result']->login_status->uid : '');
204                 $loginStatus->setOid(isset($response['result']->login_status->oid) ? $response['result']->login_status->oid : '');
205                 $loginStatus->setConnectState(isset($response['result']->login_status->connect_state) ? $response['result']->login_status->connect_state : '');
206             }
207 
208             $result['login_status'] = $loginStatus;
209 
210             return $result;
211         } catch (InvalidGrantException $e) {
212             throw new InvalidGrantException('Error [' . __FUNCTION__ . '] - Maybe "code" is reused - '.$e->getMessage());
213         } catch (Exception $e) {
214             throw new Exception('Error [' . __FUNCTION__ . '] - '.$e->getMessage());
215         }
216     }
