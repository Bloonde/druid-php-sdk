214:     /**
215:      * Returns the user data stored trough the Genetsis ID personal identifier.
216:      * The identifiers could be: id (ckusid), screenName, email, dni
217:      * Sample: array('id'=>'XXXX','screenName'=>'xxxx');
218:      *
219:      * @param array The Genetsis IDs identifier to search, 'identifier' => 'value'
220:      * @return array A vector of {@link User} objects with user's
221:      *     personal data. The array could be empty.
222:      * @throws /Exception
223:      */
224:     public static function getUsers($identifiers)
225:     {
226:         $gid_user = array();
227: 
228:         if (is_array($identifiers)) {
229:             try {
230:                 if (!$gid_user_data = FileCache::get('user-' . reset($identifiers))) {
231:                     Identity::getLogger()->debug('Identifier: ' . reset($identifiers) . ' is Not in Cache System');
232: 
233:                     $client_token = Identity::getThings()->getClientToken();
234: 
235:                     if (is_null($client_token)) {
236:                         throw new Exception('The clientToken is empty');
237:                     }
238: 
239:                     /**
240:                      * Parameters:
241:                      * oauth_token: client token
242:                      * s (select): dynamic user data to be returned
243:                      * f (from): User
244:                      * w (where): param with OR w.param1&w.param2...
245:                      */
246:                     $params = array();
247:                     $params['oauth_token'] = $client_token->getValue();
248:                     $params['s'] = "*";
249:                     $params['f'] = "User";
250:                     foreach ($identifiers as $key => $val) {
251:                         $params['w.' . $key] = $val;
252:                     }
253: 
254:                     $base = OAuthConfig::getApiUrl('api.user', 'base_url');
255:                     $api = OAuthConfig::getApiUrl('api.user', 'user');
256: 
257:                     $response = Request::execute($base . $api, $params, Request::HTTP_POST);
258: 
259:                     if (($response['code'] != 200) || (!isset($response['result']->data)) || ($response['result']->count == '0')) {
260:                         throw new Exception('The data retrieved is empty');
261:                     }
262:                     $gid_user = $response['result']->data;
263:                     FileCache::set('user-' . reset($identifiers), $response['result']->data, 3600);
264:                 } else {
265:                     Identity::getLogger()->debug('Identifier: ' . reset($identifiers) . ' is in Cache System');
266:                     $gid_user = $gid_user_data;
267:                 }
268:             } catch (Exception $e) {
269:                 Identity::getLogger()->error($e->getMessage());
