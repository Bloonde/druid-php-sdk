 65     /**
 66      * When you initialize the library, the configuration defined in "oauthconf.xml" file of the gid_client is loaded
 67      * and by default this method auto-sync data (client_token, access_token,...) with server
 68      *
 69      * @param array $settings Accepts:
 70      *      - 'app': (string) The app to load. The default app is 'default'.
 71      *      - 'xml': (string) The configuration XML.
 72      *      - 'sync': (boolean) Indicates whether to automatically synchronize data against the server. Default is TRUE.
 73      *      - 'ini_path': (string) If defined should be the internal path to the configuration file of the library (the druid.ini file).
 74      *          If this property is not defined then the library will search for this file at the default folder.
 75      *      - 'logger': (\Genetsis\core\Logger\Contracts\LoggerServiceInterface) If you want to define the logger for you application.
 76      * @return void
 77      * @throws \Exception
 78      */
 79     public static function initialize(array $settings = [])
 80     {
 81         try {
 82             if (!static::$initialized) {
 83                 if (!isset($settings['ini_path'])) {
 84                     $settings['ini_path'] = dirname(__FILE__) . '/../config/druid.ini';
 85                 }
 86                 Config::$ini_path = $settings['ini_path'];
 87                 self::$initialized = true;
 88                 AutoloaderClass::init();
 89 
 90                 // Init config library
 91                 if (!isset($settings['app'])) { $settings['app'] = 'default'; }
 92                 if (!isset($settings['sync'])) { $settings['sync'] = true; }
 93                 Config::init($settings['app'], $settings['ini_path']);
 94 
 95                 SC::setHttpService(new Http());
 96 
 97                 if (isset($settings['logger']) && ($settings['logger'] instanceof LoggerServiceInterface)) { // Custom logger.
 98                     SC::setLogger($settings['logger']);
 99                 } else { // Default logger based on configuration.
100                     if ((Config::logLevel() === 'OFF') && (!isset($_COOKIE['gidlog']))) {
101                         SC::setLogger(new EmptyLogger());
102                     } else {
103                         SC::setLogger(new DruIDLogger($_SERVER['DOCUMENT_ROOT'] . '/' . Config::logPath(), LogLevelsCollection::DEBUG));
104                     }
105                 }
106 
107                 // Initialize Cache.
108                 FileCache::init(Config::cachePath(), Config::environment());
109 
110                 // Initialize OAuth Config
111                 // TODO: check if there is a cache for the OAuth.
112                 $config = (isset($settings['xml']) && $settings['xml'])
113                     ? OAuthConfigService::buildConfigFromXml($settings['xml'], Config::CONF_VERSION)
114                     : call_user_func(function() {
115                         $file = realpath($_SERVER['DOCUMENT_ROOT'] . '/' . Config::configPath());
116                         SC::getLogger()->info('XML not defined. Trying to load from file: ' . $file, __METHOD__, __LINE__);
117                         return OAuthConfigService::buildConfigFromXmlFile($file, Config::CONF_VERSION);
118                     });
119                 SC::setOAuthService(new OAuth($config));
120 
121                 self::$gid_things = (isset($_SESSION['Things']) && (($my_things = @unserialize($_SESSION['Things'])) instanceof Things))
122                     ? $my_things
123                     : new Things();
124 
125                 if ($settings['sync']) {
126                     self::synchronizeSessionWithServer();
127                 }
128             }
129         } catch (Exception $e) {
130             SC::getLogger()->error($e, __METHOD__, __LINE__);
131             throw $e;
132         }
133     }

176     /**
177      * This method verifies the authorization tokens (client_token,
178      * access_token and refresh_token). Also updates the web client status,
179      * storing the client_token, access_token and refresh tokend and
180      * login_status in Things {@link Things}.
181      *
182      * Is INVOKE ON EACH REQUEST in order to check and update
183      * the status of the user (not logged, logged or connected), and
184      * verify that every token that you are gonna use before is going to be
185      * valid.
186      *
187      * @return void
188      */
189     public static function synchronizeSessionWithServer()
190     {
191         if (!self::$synchronized) {
192             self::$synchronized = true;
193 
194             try {
195                 SC::getLogger()->debug('Synchronizing session with server', __METHOD__, __LINE__);
196                 self::checkAndUpdateClientToken();
197 
198                 self::loadUserTokenFromPersistence();
199 
200                 if (self::$gid_things->getAccessToken() == null) {
201                     SC::getLogger()->debug('User is not logged, check SSO', __METHOD__, __LINE__);
202                     self::checkSSO();
203                     if (self::$gid_things->getRefreshToken() != null) {
204                         SC::getLogger()->debug('User not logged but has Refresh Token', __METHOD__, __LINE__);
205                         self::checkAndRefreshAccessToken();
206                     }
207                 } else {
208                     if (self::isExpired(self::$gid_things->getAccessToken()->getExpiresAt())) {
209                         SC::getLogger()->debug('User logged but Access Token is expires', __METHOD__, __LINE__);
210                         self::checkAndRefreshAccessToken();
211                     } else {
212                         SC::getLogger()->debug('User logged - check Validate Bearer', __METHOD__, __LINE__);
213                         self::checkLoginStatus();
214                     }
215                     if (!self::isConnected()) {
216                         SC::getLogger()->warn('User logged but is not connected (something wrong) - clear session data', __METHOD__, __LINE__);
217                         self::clearLocalSessionData();
218                     }
219                 }
220             } catch (Exception $e) {
221                 SC::getLogger()->error($e->getMessage(), __METHOD__, __LINE__);
222             }
223             $_SESSION['Things'] = @serialize(self::$gid_things);
224         }
225     }

227     /**
228      * Checks and updates the "client_token" and cache if we have a valid one
229      * If we don not have a Client Token in session, we check if we have a cookie
230      * If we don not have a client Token in session or in a cookie, We request a new Client Token.
231      * This method set the Client Token in Things
232      *
233      * @return void
234      * @throws \Exception
235      */
236     private static function checkAndUpdateClientToken()
237     {
238         try {
239             SC::getLogger()->debug('Checking and update client_token.', __METHOD__, __LINE__);
240             if (!(($client_token = unserialize(FileCache::get('client_token'))) instanceof ClientToken) || ($client_token->getValue() == '')) {
241                 SC::getLogger()->debug('Get Client token', __METHOD__, __LINE__);
242 
243                 if ((self::$gid_things->getClientToken() == null) || (SC::getOAuthService()->getStoredToken(TokenTypesCollection::CLIENT_TOKEN) == null)) {
244                     SC::getLogger()->debug('Not has clientToken in session or cookie', __METHOD__, __LINE__);
245 
246                     if (!$client_token = SC::getOAuthService()->getStoredToken(TokenTypesCollection::CLIENT_TOKEN)) {
247                         SC::getLogger()->debug('Token Cookie does not exists. Requesting a new one.', __METHOD__, __LINE__);
248                         $client_token = SC::getOAuthService()->doGetClientToken((string)SC::getOAuthService()->getConfig()->getEndPoint('token_endpoint'));
249                     }
250                     self::$gid_things->setClientToken($client_token);
251                 } else {
252                     SC::getLogger()->debug('Client Token from session', __METHOD__, __LINE__);
253                 }
254                 FileCache::set('client_token', serialize(self::$gid_things->getClientToken()), self::$gid_things->getClientToken()->getExpiresIn());
255             } else {
256                 SC::getLogger()->debug('Client Token from cache', __METHOD__, __LINE__);
257                 self::$gid_things->setClientToken($client_token);
258             }
259         } catch (Exception $e) {
260             throw $e;
261         }
262     }

264     /**
265      * Checks if user is logged via SSO (datr cookie) - Single Sign On
266      *
267      * The method obtain the "access_token" of the logged user in
268      * "*.cocacola.es" through the cookie, with Grant Type EXCHANGE_SESSION
269      * To SSO on domains that are not under .cocacola.es the site must include this file
270      * <script type="text/javascript" src="https://register.cocacola.es/login/sso"></script>
271      *
272      * @return void
273      * @throws /Exception
274      */
275     private static function checkSSO()
276     {
277         try {
278             $datr = call_user_func(function(){
279                 if (!isset($_COOKIE) || !is_array($_COOKIE)) {
280                     return false;
281                 }
282 
283                 if (isset($_COOKIE['datr']) && !empty($_COOKIE['datr'])) {
284                     return $_COOKIE['datr'];
285                 }
286 
287                 foreach ($_COOKIE as $key => $val) {
288                     if (strpos($key, 'datr_') === 0) {
289                         return $val;
290                     }
291                 }
292 
293                 return false;
294             });
295 
296             if ($datr) {
297                 SC::getLogger()->info('DATR cookie was found.', __METHOD__, __LINE__);
298 
299                 $response = SC::getOAuthService()->doExchangeSession(
300                     (string)SC::getOAuthService()->getConfig()->getEndPoint('token_endpoint'),
301                     $datr
302                 );
303 
304                 self::$gid_things->setAccessToken($response['access_token']);
305                 self::$gid_things->setRefreshToken($response['refresh_token']);
306                 self::$gid_things->setLoginStatus($response['login_status']);
307             } else {
308                 SC::getLogger()->debug('DATR cookie not exist, user is not logged', __METHOD__, __LINE__);
309             }
310         } catch (InvalidGrantException $e) {
311             unset($_COOKIE[OAuth::SSO_COOKIE_NAME]);
312             setcookie(OAuth::SSO_COOKIE_NAME, null, -1, null);
313 
314             SC::getLogger()->warn('Invalid Grant, check an invalid DATR', __METHOD__, __LINE__);
315             throw $e;
316         } catch (Exception $e) {
317             throw $e;
318         }
319     }

363     {

376                     unset($_SESSION[$key]);
377                 }
378             }
379         }
